\end{enumerate}

\section{Data}
\begin{frame}
\frametitle{Reproduction procedure}
\vspace{0.1in}
\begin{footnotesize}
\textbf{Replication lab}: undergraduates over 5 summers (2014-2018)
\begin{itemize}
\item Minimum knowledge: economics, computer sciences or Operations Research
\item Training: version control systems, cloud computing, reproduction procedure
\end{itemize}
\vspace{0.1in}

\textbf{3-step process}:
\begin{enumerate}
\item<1-> Entry questionnaire

\begin{itemize}
\item<1-> \footnotesize{Descriptive information: online appendix, data sets presence and availability, programs, documentation and its clarity}
\item<1-> \footnotesize{Expected level of replication difficulty}
\end{itemize}
\break
\item<2-> Reproduction exercise: record any change in VCS, replication summary in a txt
\item<3-> Exit questionnaire
\begin{itemize}
\item<3-> \footnotesize{Reproduction success, reasons for failure}
\end{itemize}

\end{enumerate}
\end{footnotesize}
\end{frame}
